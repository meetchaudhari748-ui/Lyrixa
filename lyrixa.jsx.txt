// ─── STUDIO ───────────────────────────────────────────────────────────────
function Studio() {
  const [tool,setTool]=useState(0);
  const [input,setInput]=useState("");
  const [mood,setMood]=useState(TOOLS[0].moods[0]);
  const [lang,setLang]=useState("Hindi");
  const [output,setOutput]=useState("");
  const [loading,setLoading]=useState(false);
  const [singing,setSinging]=useState(false);
  const [copied,setCopied]=useState(false);
  const [used,setUsed]=useState(0);
  const FREE=5;
  const T=TOOLS[tool];

  const switchTool=i=>{setTool(i);setInput("");setMood(TOOLS[i].moods[0]);setOutput("");window.speechSynthesis?.cancel();setSinging(false);};

  const generate=async()=>{
    if(!input.trim())return;
    if(used>=FREE){setOutput("⚠️ Free limit (5/day) khatam! Starter plan lo — sirf ₹159/month. Unlimited access milega.");return;}
    setLoading(true);setOutput("");setSinging(false);window.speechSynthesis?.cancel();
    try{
      const res=await fetch("https://api.anthropic.com/v1/messages",{
        method:"POST",headers:{"Content-Type":"application/json"},
        body:JSON.stringify({model:"claude-sonnet-4-20250514",max_tokens:1000,
          messages:[{role:"user",content:T.prompt(input,mood,lang)}]}),
      });
      const data=await res.json();
      setOutput(data.content?.map(b=>b.text||"").join("")||"Error aaya, retry karo.");
      setUsed(u=>u+1);
    }catch{setOutput("❌ Connection error. Internet check karo.");}
    setLoading(false);
  };

  const sing=()=>{
    if(!output||output.startsWith("⚠️")||output.startsWith("❌"))return;
    if(!("speechSynthesis" in window)){alert("Chrome ya Edge browser use karo — voice support chahiye!");return;}
    if(singing){window.speechSynthesis.cancel();setSinging(false);return;}
    window.speechSynthesis.cancel();
    setSinging(true);
    const moodMap={"Sad 💔":{r:.72,p:.75},"Happy 🎉":{r:1.05,p:1.35},"Romantic 💕":{r:.8,p:1.1},"Angry 😤":{r:1.15,p:.6},"Motivational 🔥":{r:1.0,p:1.25},"Devotional 🙏":{r:.65,p:.85},"Party 🎊":{r:1.1,p:1.4},"Chill 😎":{r:.75,p:1.0}};
    const m=moodMap[mood]||{r:.9,p:1.0};
    const doSpeak=(voices)=>{
      const u=new SpeechSynthesisUtterance(output.replace(/\[.*?\]/g,"").slice(0,700));
      u.rate=m.r; u.pitch=m.p; u.volume=1;
      if(voices&&voices.length){
        const pick=voices.find(v=>v.lang==="hi-IN")||voices.find(v=>v.lang.startsWith("hi"))||voices.find(v=>v.lang==="en-IN")||voices.find(v=>v.lang.startsWith("en"))||voices[0];
        if(pick)u.voice=pick;
      }
      u.onend=()=>setSinging(false);
      u.onerror=()=>setSinging(false);
      window.speechSynthesis.speak(u);
    };
    // try immediately first
    const voices=window.speechSynthesis.getVoices();
    if(voices&&voices.length>0){doSpeak(voices);}
    else{
      // wait for voices to load
      window.speechSynthesis.onvoiceschanged=()=>{
        doSpeak(window.speechSynthesis.getVoices());
        window.speechSynthesis.onvoiceschanged=null;
      };
      // also try after short delay as fallback
      setTimeout(()=>{
        const v2=window.speechSynthesis.getVoices();
        doSpeak(v2);
      },300);
    }
  };

  const copy=()=>{navigator.clipboard.writeText(output);setCopied(true);setTimeout(()=>setCopied(false),2000);};

  return (
    <div style={{maxWidth:760,margin:"0 auto",padding:"32px 16px 60px",
      position:"relative",zIndex:5,fontFamily:"'Rajdhani',sans-serif"}}>

      {/* Header */}
      <div style={{textAlign:"center",marginBottom:28}}>
        <div style={{fontFamily:"'Share Tech Mono',monospace",fontSize:10,color:"#f7931a88",
          letterSpacing:4,marginBottom:6}}>// AI CONTENT GENERATOR</div>
        <h1 style={{fontFamily:"'Orbitron',sans-serif",fontSize:"clamp(22px,5vw,38px)",
          fontWeight:800,letterSpacing:3,color:"#e8e0d5",marginBottom:10}}>
          CONTENT <span style={{color:"#f7931a"}}>STUDIO</span>
        </h1>
        {/* Usage bar */}
        <p style={{fontFamily:"'Share Tech Mono',monospace",fontSize:10,color:"#333355",
          letterSpacing:2,marginBottom:8}}>{FREE-used} FREE GENERATIONS REMAINING TODAY</p>
        <div style={{display:"flex",gap:5,justifyContent:"center"}}>
          {Array.from({length:FREE}).map((_,i)=>(
            <div key={i} style={{width:30,height:3,borderRadius:2,
              background:i<used?"#f7931a":"#1a1a30",transition:"background .3s"}}/>
          ))}
        </div>
      </div>

      {/* Tool tabs */}
      <div style={{display:"flex",gap:6,marginBottom:20,overflowX:"auto",paddingBottom:4}}>
        {TOOLS.map((t,i)=>(
          <button key={t.id} className={`tool-tab ${tool===i?"active":""}`} onClick={()=>switchTool(i)}>
            {t.emoji} {t.label}
          </button>
        ))}
      </div>

      {/* Input card */}
      <div className="cyber-card" style={{padding:"22px 20px",marginBottom:14}}>
        {/* Language */}
        <div style={{fontFamily:"'Share Tech Mono',monospace",fontSize:9,color:"#f7931a88",
          letterSpacing:3,textTransform:"uppercase",marginBottom:8}}>// LANGUAGE</div>
        <div style={{marginBottom:18}}>
          <LangPicker value={lang} onChange={setLang}/>
        </div>

        {/* Input */}
        <div style={{fontFamily:"'Share Tech Mono',monospace",fontSize:9,color:"#f7931a88",
          letterSpacing:3,textTransform:"uppercase",marginBottom:8}}>// TOPIC / IDEA</div>
        <textarea rows={3} placeholder={T.placeholder} value={input}
          onChange={e=>setInput(e.target.value)}
          onKeyDown={e=>e.key==="Enter"&&e.ctrlKey&&generate()}
          style={{marginBottom:18}}/>

        {/* Mood */}
        <div style={{fontFamily:"'Share Tech Mono',monospace",fontSize:9,color:"#f7931a88",
          letterSpacing:3,textTransform:"uppercase",marginBottom:8}}>// MOOD / STYLE</div>
        <div style={{display:"flex",gap:6,flexWrap:"wrap",marginBottom:22}}>
          {T.moods.map(m=>(
            <button key={m} className={`mood-chip ${mood===m?"on":"off"}`}
              onClick={()=>setMood(m)}>{m}</button>
          ))}
        </div>

        {/* Generate */}
        <button className="btn-primary" onClick={generate}
          disabled={loading||!input.trim()}
          style={{width:"100%",padding:"15px",fontSize:13,borderRadius:10}}>
          {loading
            ?<span style={{display:"flex",alignItems:"center",justifyContent:"center",gap:10}}>
               <span style={{width:14,height:14,border:"2px solid #00000044",borderTop:"2px solid #000",
                 borderRadius:"50%",display:"inline-block",animation:"spin 1s linear infinite"}}/>
               GENERATING...
             </span>
            :`▶ GENERATE ${T.label.toUpperCase()} IN ${lang.toUpperCase()}`}
        </button>
        <div style={{textAlign:"center",fontFamily:"'Share Tech Mono',monospace",fontSize:9,
          color:"#1a1a30",marginTop:6,letterSpacing:2}}>CTRL+ENTER SHORTCUT</div>
      </div>

      {/* Output card */}
      {(output||loading)&&(
        <div className="cyber-card fade-up" style={{padding:"22px 20px"}}>
          <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",marginBottom:14}}>
            <div style={{display:"flex",alignItems:"center",gap:10}}>
              <span style={{fontFamily:"'Share Tech Mono',monospace",fontSize:9,
                color:"#f7931a",letterSpacing:3}}>{T.emoji} OUTPUT // {lang.toUpperCase()}</span>
              {singing&&(
                <div style={{display:"flex",alignItems:"flex-end",gap:1.5,height:16}}>
                  {[1,1.8,2.5,2,1.2].map((h,i)=>(
                    <div key={i} className="wbar" style={{height:h*5,animationDelay:`${i*.1}s`}}/>
                  ))}
                </div>
              )}
            </div>
            {output&&!loading&&(
              <div style={{display:"flex",gap:8}}>
                <button onClick={sing} style={{
                  padding:"5px 12px",borderRadius:6,
                  border:`1px solid ${singing?"#a855f7":"#1a1a30"}`,
                  background:singing?"#a855f715":"transparent",
                  color:singing?"#a855f7":"#333355",
                  fontFamily:"'Share Tech Mono',monospace",fontSize:10,
                  cursor:"pointer",transition:"all .2s",letterSpacing:1,
                }}>{singing?"⏹ STOP":"🔊 AI SING"}</button>
                <button onClick={copy} style={{
                  padding:"5px 12px",borderRadius:6,
                  border:"1px solid #1a1a30",background:"transparent",
                  color:copied?"#22c55e":"#333355",
                  fontFamily:"'Share Tech Mono',monospace",fontSize:10,
                  cursor:"pointer",transition:"all .2s",letterSpacing:1,
                }}>{copied?"✓ COPIED":"📋 COPY"}</button>
              </div>
            )}
          </div>

          {loading
            ?<div style={{color:"#1a1a30",fontSize:13,fontFamily:"'Share Tech Mono',monospace",
               letterSpacing:2,animation:"pulse 1.5s infinite"}}>
               ▶ AI GENERATING CONTENT...
             </div>
            :<div style={{whiteSpace:"pre-wrap",lineHeight:1.9,fontSize:15,color:"#ccccdd",
               fontFamily:"'Noto Sans Devanagari','Rajdhani',sans-serif",fontWeight:500}}>
               {output}
             </div>
          }

          {output&&!loading&&!output.startsWith("⚠️")&&!output.startsWith("❌")&&(
            <div style={{marginTop:14,padding:"10px 14px",background:"#a855f70a",
              border:"1px solid #a855f722",borderRadius:8}}>
              <p style={{fontFamily:"'Share Tech Mono',monospace",fontSize:10,
                color:"#a855f7",letterSpacing:1}}>
                🔊 AI SING: Press button — AI will read output aloud in mood-matched voice!
              </p>
            </div>
          )}
        </div>
      )}
    </div>
  );
}
